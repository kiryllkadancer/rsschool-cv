```python
def second_index(text: str, symbol: str) -> [int, None]:
    """
        returns the second index of a symbol in a given text
    """
    new_text = text[text.find(symbol) + 1:]
    if symbol in new_text:
        return new_text.find(symbol) + (len(text) - len(new_text))
```
